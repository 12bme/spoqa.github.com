def emailenc(email_addr):
    length = 4
    tmp = email_addr.split('@')
    prefix = tmp[0]
    suffix = tmp[1]
    
    print 'prefix : ', prefix
    print 'suffix : ', suffix

    if len(prefix) > length:
        remain = len(prefix) - length
        l = list(prefix)

        for i in range(remain):
            l[length + i] = '*'

        prefix = ''.join(l)

    addr = prefix + '@' + suffix
    
    print addr

    return addr
